echo [Palavra/Frase]                     | Faz o cmd falar o que você digitar após o comando.
echo [Palavra/Frase] > [nomeDoArquivo]   | Faz o cmd redirecionar o que você digitou para um arquivo.
>——Diretórios—————————————————————————————
cd [Nome da pasta]                       | "Change Directory", Muda para a pasta especificada.
cd ..                                    | Volta para a pasta anterior.
dir                                      | "DIRectory", Mostra os arquivos e pastas na pasta atual.
del [nomeDoArquivo]                      | "DELete", deleta apenas arquivos. Use * para todos os arquivos. Use a flag /q para não aparecer a mensagem de confirmação. 
mkdir [nomeDaPasta]                      | "MaKe DIRectory", Cria uma pasta.
rmdir [nomeDaPasta] /s                   | "ReMove DIRectory", deleta apenas pastas. Use * para todas as pastas. Use a flag /q para não aparecer a mensagem de confirmação.
>——Utilitários————————————————————————————
cls                                      | "CLear Screen", Limpa o cmd.
exit                                     | Fecha o cmd.
>——Teclas—————————————————————————————————
TAB                                      | Lista, de acordo com o que escreveu, os arquivos/pastas de uma pasta. Use Shift+TAB para a direção contrária.
↑/↓                                      | Lista, de um em um, para cima ou para baixo, o histórico de comandos feitos.
>——Atalhos————————————————————————————————
Ctrl + Insert                            | Copia o texto selecionado
Shift + Insert                           | Cola o texto copiado
